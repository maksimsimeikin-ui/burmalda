        function getRandomItem() {
        const items = casesData[currentCaseId].items;
        const rand = Math.random() * 100;

        const blues = items.filter(i => i.rarity === 'blue');
        const purples = items.filter(i => i.rarity === 'purple');
        const pinks = items.filter(i => i.rarity === 'pink');
        const reds = items.filter(i => i.rarity === 'red');
        const golds = items.filter(i => i.rarity === 'gold');

        if (rand < 75 && blues.length > 0) return blues[Math.floor(Math.random() * blues.length)];
        if (rand < 90 && purples.length > 0) return purples[Math.floor(Math.random() * purples.length)];
        if (rand < 95 && pinks.length > 0) return pinks[Math.floor(Math.random() * pinks.length)];
        if (rand < 99 && reds.length > 0) return reds[Math.floor(Math.random() * reds.length)];
        if (golds.length > 0) return golds[Math.floor(Math.random() * golds.length)];

        return items[Math.floor(Math.random() * items.length)];
        }

        function trackRouletteTicks(track) {
        let lastItem = -1;
        const itemWidth = 150;

        function check() {
        if (!isOpening) return;
        const transform = window.getComputedStyle(track).getPropertyValue('transform');
        if (transform !== 'none') {
        const matrix = transform.match(/^matrix(?:3d)?\((.+)\)$/);
        if (matrix) {
        const values = matrix[1].split(', ');
        let currentX = (values.length === 16) ? Math.abs(parseFloat(values[12])) : Math.abs(parseFloat(values[4]));
        const containerWidth = track.parentElement.offsetWidth;
        const currentItem = Math.floor((currentX + containerWidth / 2) / itemWidth);
        if (currentItem !== lastItem && lastItem !== -1) {
        playTick();
        }
        lastItem = currentItem;
        }
        }
        requestAnimationFrame(check);
        }
        requestAnimationFrame(check);
        }

        if (upgMyItem && upgMyItem.id === id) {
        upgMyItem = null;
        if(document.getElementById('upgrade-page').classList.contains('active')) {
        renderUpgradeUI();
        }
        }
        if (audioUnlocked) {
        coinSound.currentTime = 0;
        coinSound.play().catch(()=>{});
        }
        }
        };

        // --- 8. СИСТЕМА АПГРЕЙДОВ ---
        function renderUpgradeUI() {
        const slotMy = document.getElementById('upg-slot-my');
        if (upgMyItem) {
        slotMy.className = `upg-slot filled rarity-${upgMyItem.rarity}`;
        slotMy.innerHTML = `
        <img src="${upgMyItem.img}">
        <div class="upg-slot-name">${upgMyItem.name}</div>
        <div class="upg-slot-price">$${upgMyItem.price.toFixed(2)}</div>
        `;
        } else {
        slotMy.className = 'upg-slot';
        slotMy.innerHTML = '<span style="color: #777;">Выберите из инвентаря</span>';
        }

        const slotTarget = document.getElementById('upg-slot-target');
        if (upgTargetItem) {
        slotTarget.className = `upg-slot filled rarity-${upgTargetItem.rarity}`;
        slotTarget.innerHTML = `
        <img src="${upgTargetItem.img}">
        <div class="upg-slot-name">${upgTargetItem.name}</div>
        <div class="upg-slot-price">$${upgTargetItem.price.toFixed(2)}</div>
        `;
        } else {
        slotTarget.className = 'upg-slot';
        slotTarget.innerHTML = '<span style="color: #777;">Выберите из пула</span>';
        }

        let chance = 0;
        if (upgMyItem && upgTargetItem) {
        chance = (upgMyItem.price / upgTargetItem.price) * 100;
        if (chance > 100) chance = 100;
        }

        document.getElementById('upg-chance-text').innerText = (chance > 0) ? chance.toFixed(1) + '%' : '0%';
        const deg = chance * 3.6;
        document.getElementById('upg-wheel').style.background = `conic-gradient(#4caf50 0deg, #4caf50 ${deg}deg, #1e1e24 ${deg}deg, #1e1e24 360deg)`;

        const btn = document.getElementById('btn-do-upgrade');
        btn.disabled = !(upgMyItem && upgTargetItem);

        renderUpgradeLists();
        }

        function trackUpgradeTicks(arrow) {
        let lastSector = -1;
        function check() {
        if (!isUpgrading) return;
        const transform = window.getComputedStyle(arrow).getPropertyValue('transform');
        if (transform !== 'none') {
        const matrix = transform.match(/^matrix(?:3d)?\((.+)\)$/);
        if (matrix) {
        const values = matrix[1].split(', ');
        const a = parseFloat(values[0]);
        const b = parseFloat(values[1]);
        let angle = Math.round(Math.atan2(b, a) * (180/Math.PI));
        if (angle < 0) angle += 360;
        const currentSector = Math.floor(angle / 15);
        if (currentSector !== lastSector && lastSector !== -1) {
        playTick();
        }
        lastSector = currentSector;
        }
        }
        requestAnimationFrame(check);
        }
        requestAnimationFrame(check);
        }

        window.doUpgrade = function() {
        if (!upgMyItem || !upgTargetItem || isUpgrading) return;

        const finalChance = (upgMyItem.price / upgTargetItem.price) * 100;
        const btn = document.getElementById('btn-do-upgrade');
        const msg = document.getElementById('upg-result-msg');
        const arrow = document.getElementById('upg-arrow');

        btn.disabled = true;
        msg.style.color = "#ffae42";
        msg.innerText = "Крутим рулетку...";

        const indexToRemove = inventory.findIndex(i => i.id === upgMyItem.id);
        if (indexToRemove > -1) {
        inventory.splice(indexToRemove, 1);
        saveData();
        }

        const roll = Math.random() * 100;
        const isWin = roll <= finalChance;

        const stopAngle = roll * 3.6;
        const extraSpins = 5 * 360;

        arrowCurrentRotation += extraSpins + stopAngle - (arrowCurrentRotation % 360);

        arrow.style.transition = 'transform 3.5s cubic-bezier(0.15, 0.85, 0.15, 1)';
        arrow.style.transform = `rotate(${arrowCurrentRotation}deg)`;

        isUpgrading = true;
        trackUpgradeTicks(arrow);

        setTimeout(() => {
        isUpgrading = false;
        if (isWin) {
        playWin();
        msg.style.color = "#4caf50";
        msg.innerText = "УСПЕШНЫЙ АПГРЕЙД!";
        const wonItem = { ...upgTargetItem, id: Date.now() + Math.random() };
        inventory.push(wonItem);
        saveData();
        } else {
        msg.style.color = "#eb4b4b";
        msg.innerText = "АПГРЕЙД ПРОВАЛЕН :(";
        }

        upgMyItem = null;
        setTimeout(() => {
        renderUpgradeUI();
        }, 2000);
        }, 3600);
        }
